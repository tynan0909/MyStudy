5장 WHERE절과 연산자

WHERE는 FROM보다 좁은곳에서 조회하는 명령어임
EX) SELECT *
	FROM EMP
	WHERE DEPTNO = 30; (전체에서 EMP뽑아놓고 사번이 30인 녀석) 자바와 다르게 =가 동등연산자임
숫자는 그냥 비교하면 되는데 문자는 ' '를 놔야 함

~~WHERE DEPTNO = 30
	AND JOB = 'SALESMAN';(사번이 30인놈들중 JOB이 세일즈맨인놈)

~~WHERE DEPTNO = 30
	OR JOB = 'CLERK'; (사번이 30이거나 JOB이 CLERK인놈)

FROM EMP
WHERE SAL*12 = 36000; (월급X12가 36000인놈)
>= <=도 가능 (이상 이하)

WHERE ENAME >= 'F'; (이름앞자리가 F보다 큰사람)[F의 뒷자리]
